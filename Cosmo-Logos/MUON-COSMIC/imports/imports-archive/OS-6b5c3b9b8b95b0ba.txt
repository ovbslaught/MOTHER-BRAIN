@echo off
setlocal enabledelayedexpansion

REM ==============================================================================
REM Cosmic Key OS - VultureCode Master Protocol (VMP) - Production Ready
REM This script automates the full operational cycle for a Windows system.
REM It is designed for seamless, no-manual-labor execution.
REM ==============================================================================

REM --- GLOBAL CONFIGURATION ---
REM Adjust these variables for your specific local system setup.
set "PRIMARY_DRIVE=F:"
set "COSMIC_KEY_ROOT=%PRIMARY_DRIVE%\CosmicKey"
set "VULTURE_CODE_DIR=%COSMIC_KEY_ROOT%\VultureCode"
set "PROJECTS_DIR=%COSMIC_KEY_ROOT%\Projects"
set "SNAPSHOTS_DIR=%COSMIC_KEY_ROOT%\Snapshots"
set "ASSETS_DIR=%COSMIC_KEY_ROOT%\Assets\Visuals"
set "LOG_FILE=%VULTURE_CODE_DIR%\vmp_production.log"
set "REPO_URL=https://github.com/ovbslaught/cosmickey.git"
set "BRANCH_LIST_FILE=%VULTURE_CODE_DIR%\unlocked_branches.txt"

REM --- CORE FUNCTIONS ---
REM Provides a time-stamped, status-flagged logging.
:log_message
    set "type=%~1"
    set "message=%~2"
    echo %date% %time% [!type!]: !message! >> "%LOG_FILE%"
goto :EOF

REM Verifies and creates all necessary directories.
:verify_directories
    call :log_message INFO "ARCHON Protocol: Verifying system directories."
    if not exist "%COSMIC_KEY_ROOT%" (
        call :log_message INFO "Creating root directory."
        mkdir "%COSMIC_KEY_ROOT%"
    )
    if not exist "%VULTURE_CODE_DIR%" mkdir "%VULTURE_CODE_DIR%"
    if not exist "%PROJECTS_DIR%" mkdir "%PROJECTS_DIR%"
    if not exist "%SNAPSHOTS_DIR%" mkdir "%SNAPSHOTS_DIR%"
    if not exist "%ASSETS_DIR%" mkdir "%ASSETS_DIR%"
    call :log_message SUCCESS "All directories verified."
goto :EOF

REM Handles the full Git synchronization.
:run_git_sync
    call :log_message TASK "VultureCode Protocol: Running Git Sync."
    if not exist "%COSMIC_KEY_ROOT%\.git" (
        call :log_message INFO "Cloning fresh repository."
        git clone "%REPO_URL%" "%COSMIC_KEY_ROOT%" >> "%LOG_FILE%" 2>&1
        if !errorlevel! neq 0 (
            call :log_message ERROR "Git clone failed. Aborting."
            exit /b 1
        )
    )
    cd "%COSMIC_KEY_ROOT%"
    git pull origin main >> "%LOG_FILE%" 2>&1
    if !errorlevel! neq 0 (
        call :log_message ERROR "Git pull failed. Continuing with caution."
    )
    git add .
    git commit -m "VMP auto-run %date% %time%" >> "%LOG_FILE%" 2>&1
    git push origin main >> "%LOG_FILE%" 2>&1
    if !errorlevel! neq 0 (
        call :log_message ERROR "Git push failed. Aborting."
        exit /b 1
    )
    call :log_message SUCCESS "Git synchronization complete."
goto :EOF

REM The Sentinel Protocol dynamically generates a new mission.
:generate_mission_card
    call :log_message TASK "Sentinel Protocol: Generating mission card."
    set "branches=main_canon cope_files xurator_tales_from_the_shadow_realm interstellar_branch sol's_journey bytez_jokebook nodes_diary echoes"
    set "mission_card=%PROJECTS_DIR%\mission_card_%TIMESTAMP%.txt"

    set "branches_arr[0]=main_canon"
    set "branches_arr[1]=cope_files"
    set "branches_arr[2]=xurator_tales_from_the_shadow_realm"
    set "branches_arr[3]=interstellar_branch"
    set "branches_arr[4]=sol's_journey"
    set "branches_arr[5]=bytez_jokebook"
    set "branches_arr[6]=nodes_diary"
    set "branches_arr[7]=echoes"

    set /a "rand_num=!RANDOM! %% 8"
    set "selected_branch=!branches_arr[%rand_num%]!"

    echo --- NOMADZ CARD: MISSION DEPLOYED --- > "%mission_card%"
    echo BRANCH_ACCESS: !selected_branch! >> "%mission_card%"
    echo OBJECTIVE: Investigate an anomaly in the selected branch. >> "%mission_card%"
    echo TEAM: Orginz Branch >> "%mission_card%"
    echo PRIORITY: HIGH >> "%mission_card%"
    echo --- END NOMADZ CARD --- >> "%mission_card%"

    call :log_message SUCCESS "New mission generated for branch: !selected_branch!."
goto :EOF

REM Archives a placeholder visual asset.
:archive_visuals
    call :log_message TASK "Echo Prime: Archiving visual assets."
    set "asset_url=https://oaidalleapiprodscus.blob.core.windows.net/private/images/Nomadz_Mission1.jpg"
    set "asset_name=nomadz_mission_%TIMESTAMP%.jpg"
    powershell -Command "Invoke-WebRequest -Uri '%asset_url%' -OutFile '%ASSETS_DIR%\%asset_name%'" >> "%LOG_FILE%" 2>&1
    call :log_message SUCCESS "Visual asset archived: !asset_name!."
goto :EOF

REM Creates a full system snapshot.
:create_snapshot
    call :log_message TASK "Mother Brain: Creating system snapshot."
    set "snapshot_name=cosmic_snapshot_%TIMESTAMP%.zip"
    powershell -Command "Add-Type -A 'System.IO.Compression.FileSystem'; [IO.Compression.ZipFile]::CreateFromDirectory('%COSMIC_KEY_ROOT%', '%SNAPSHOTS_DIR%\%snapshot_name%')" >> "%LOG_FILE%" 2>&1
    call :log_message SUCCESS "Snapshot created: !snapshot_name!."
goto :EOF

REM --- MAIN EXECUTION BLOCK ---
REM This is the entry point for the script.
(
    REM Generate a unified timestamp for the entire run.
    for /f "tokens=2 delims==" %%i in ('"wmic os get localdatetime /value"') do set "ldt=%%i"
    set TIMESTAMP=%ldt:~0,8%-%ldt:~8,6%
    echo %TIMESTAMP% [START]: VMP Protocol Initiated. > "%LOG_FILE%"
    
    call :verify_directories
    call :run_git_sync
    call :generate_mission_card
    call :archive_visuals
    call :create_snapshot

    call :log_message SUCCESS "VMP Protocol completed successfully."
    echo.
    echo ==========================================================
    echo VMP Protocol completed. See log for details: %LOG_FILE%
    echo ==========================================================
)
goto :EOF
