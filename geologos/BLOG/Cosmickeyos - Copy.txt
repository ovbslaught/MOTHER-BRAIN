# Utility Functions

def ensure_folder(path):
    if not os.path.exists(path):
        os.makedirs(path)

def backup_cosmickey():
    ensure_folder(BACKUP_ROOT)
    timestamp = time.strftime("%Y%m%d_%H%M%S")
    backup_path = os.path.join(BACKUP_ROOT, f"CosmicKeyBackup_{timestamp}")
    shutil.copytree(COSMIC_KEY_ROOT, backup_path, dirs_exist_ok=True)
    print(f"[Backup] Cosmic Key backed up to {backup_path}")

def extract_all_zips(root):
    for folder, _, files in os.walk(root):
        for f in files:
            if f.endswith(".zip"):
                zip_path = os.path.join(folder, f)
                extract_path = os.path.join(folder, f"extracted_{f[:-4]}")
                with zipfile.ZipFile(zip_path, 'r') as zip_ref:
                    zip_ref.extractall(extract_path)
                os.remove(zip_path)

def download_engine(name, url):
    path = os.path.join(ENGINES_ROOT, name)
    if not os.path.exists(path):
        ensure_folder(path)
        zip_path = os.path.join(path, f"{name}.zip")
        print(f"[Engine] Downloading {name}...")
        r = requests.get(url, stream=True)
        with open(zip_path, "wb") as f:
            shutil.copyfileobj(r.raw, f)
        with zipfile.ZipFile(zip_path, 'r') as zip_ref:
            zip_ref.extractall(path)
        os.remove(zip_path)
        print(f"[Engine] {name} ready.")
    else:
        print(f"[Engine] {name} already exists.")

def generate_nomadz_card():

class UniverseVisualization(QGraphicsView):
    def __init__(self, parent=None):
        super().__init__(parent)
        self.scene = QGraphicsScene()
        self.setScene(self.scene)
        self.setRenderHint(self.RenderHint.Antialiasing)
        self.realms = {}
        self.wormholes = []
        self.missions = []
        self.timer = QTimer()
        self.timer.timeout.connect(self.update_universe)
        self.timer.start(2000)

    def generate_universe(self):
        self.scene.clear()
        self.realms.clear()
        self.wormholes.clear()
        self.missions.clear()
        width = 800
        height = 400
        num_realms = random.randint(5, 10)
        for i in range(num_realms):
            x = random.randint(50, width)
            y = random.randint(50, height)
            color = QColor(random.randint(50,255), random.randint(50,255), random.randint(50,255))
            ellipse = self.scene.addEllipse(x, y, 30, 30, QPen(Qt.black), QBrush(color))
            self.realms[f"Realm_{i}"] = (ellipse, (x+15,y+15))
        realm_keys = list(self.realms.keys())
        for i in range(len(realm_keys)-1):
            r1 = self.realms[realm_keys[i]][1]
            r2 = self.realms[realm_keys[i+1]][1]
            line = self.scene.addLine(r1[0], r1[1], r2[0], r2[1], QPen(Qt.white,2))
            self.wormholes.append((line, r1, r2))
        for line, start, end in self.wormholes:
            mission = self.scene.addEllipse(start[0]-5, start[1]-5, 10,10, QPen(Qt.yellow), QBrush(Qt.yellow))
            self.missions.append({"item": mission, "start": start, "end": end, "progress": 0})

    def update_universe(self):
        if not self.realms:
            self.generate_universe()
            return
        for mission in self.missions:
            mission["progress"] += 0.05
            if mission["progress"] > 1.0:
                mission["progress"] = 0
            x = mission["start"][0] + (mission["end"][0]-mission["start"][0])*mission["progress"]
            y = mission["start"][1] + (mission["end"][1]-mission["start"][1])*mission["progress"]
            mission["item"].setPos(x-5,y-5)
