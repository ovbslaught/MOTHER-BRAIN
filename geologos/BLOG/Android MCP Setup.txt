# ANDROID MCP BRIDGE SETUP
# Galaxy S24 + Termux + MCP Servers Integration

## 1. Install Termux on Galaxy S24
1. Download Termux from F-Droid (NOT Play Store)
2. Install Termux:API plugin from F-Droid
3. Grant storage and location permissions

## 2. Copy Commands to Termux

```bash
# Update packages
pkg update && pkg upgrade -y

# Install required packages
pkg install -y python git nodejs npm curl wget nano termux-api

# Create development directories
mkdir -p ~/Development/{Tools,Scripts,Workspace}

# Install MCP servers
pip install mcp-server mcp-client httpx websockets

# Clone MCP server repositories
cd ~/Development/Tools
git clone https://github.com/modelcontextprotocol/servers.git mcp-servers
git clone https://github.com/spdustin/ChatGPT-to-MCP-bridge.git
git clone https://github.com/modelcontextprotocol/inspect-ai.git inspect-ai

# Install Node.js dependencies
cd ~/Development/Tools/ChatGPT-to-MCP-bridge
npm install

# Create Android MCP configuration
mkdir -p ~/.config/mcp
cat > ~/.config/mcp/android_config.json << 'EOF'
{
  "mcpServers": {
    "android-bridge": {
      "command": "python",
      "args": ["~/Development/Tools/mcp-servers/src/android/android_server.py"],
      "env": {
        "ANDROID_SDK_ROOT": "/system",
        "TERMUX_VERSION": "1.0"
      }
    },
    "filesystem": {
      "command": "python", 
      "args": ["~/Development/Tools/mcp-servers/src/filesystem/filesystem_server.py"],
      "env": {
        "ALLOWED_DIRECTORIES": "/sdcard,/storage/emulated/0,/data/data/com.termux/files/home"
      }
    },
    "termux-api": {
      "command": "python",
      "args": ["~/Development/Tools/mcp-servers/src/termux/termux_server.py"],
      "env": {
        "TERMUX_API_ENABLED": "true"
      }
    }
  }
}
EOF

# Create startup script
cat > ~/Development/Scripts/start_mcp_bridge.sh << 'EOF'
#!/bin/bash
echo "Starting Android MCP Bridge..."

# Start MCP servers in background
cd ~/Development/Tools/mcp-servers
python -m mcp.server.filesystem --allowed-directories /sdcard,/storage/emulated/0 ~/ &
python -m mcp.server.android &

# Start ChatGPT bridge
cd ~/Development/Tools/ChatGPT-to-MCP-bridge
npm start &

# Start Termux API bridge
cd ~/Development/Tools/mcp-servers
python src/termux/termux_server.py &

echo "All MCP services started in background"
echo "Connect to: http://localhost:3000"
echo "MCP endpoints available:"
echo "  - Filesystem: http://localhost:3001"
echo "  - Android API: http://localhost:3002" 
echo "  - Termux API: http://localhost:3003"
EOF

chmod +x ~/Development/Scripts/start_mcp_bridge.sh

# Create Termux boot script
cat > ~/.bashrc << 'EOF'
# Android MCP Bridge Auto-start
if [ -f ~/Development/Scripts/start_mcp_bridge.sh ]; then
    ~/Development/Scripts/start_mcp_bridge.sh
fi

# Custom PATH
export PATH="$PATH:~/Development/Tools:~/Development/Scripts:~/Development/Workspace"

# MCP Environment
export MCP_CONFIG_PATH="$HOME/.config/mcp"
export MCP_SERVER_PORT=3000
export MCP_FILESYSTEM_PORT=3001
export MCP_ANDROID_PORT=3002
export MCP_TERMUX_PORT=3003
EOF

echo "Android MCP Bridge setup complete!"
echo "Restart Termux to activate configuration"
```

## 4. Test Connection

From Termux:
```bash
# Test MCP bridge
~/Development/Scripts/start_mcp_bridge.sh

# Test filesystem access
curl http://localhost:3001/filesystem/list?path=/sdcard

# Test Android API
curl http://localhost:3002/android/info

# Test Termux API
curl http://localhost:3003/termux/battery

# Get Android IP for Windows connection
termux-wifi-connectioninfo
```

From Windows:
```cmd
# Test connection to Android bridge
curl http://YOUR_ANDROID_IP:3001/filesystem/list?path=/sdcard
curl http://YOUR_ANDROID_IP:3002/android/info
curl http://YOUR_ANDROID_IP:3003/termux/battery
```

## 5. Claude Desktop Integration

Add to Claude Desktop config:
```json
{
  "mcpServers": {
    "android-bridge": {
      "command": "curl",
      "args": ["http://YOUR_ANDROID_IP:3000/mcp"]
    }
  }
}
```

Replace YOUR_ANDROID_IP with your Galaxy S24's IP address.
