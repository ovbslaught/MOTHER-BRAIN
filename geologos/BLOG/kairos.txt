# --- Tool Identification Heuristics (Copied from index.html for data enrichment) ---
def identify_tools(script):
    """Identifies tools used in the script for logging metrics."""
    toolRegex = r"(?:pkg|apt|pip|npm) install\s+([a-zA-Z0-9-_\s]+)"
    matches = list(set(re.findall(toolRegex, script)))
    
    tools = set()
    for match in matches:
        match.strip().split().forEach(tool => tools.add(tool))

    # Heuristics for common built-ins
    if 'curl' in script: tools.add('curl')
    if 'wget' in script: tools.add('wget')
    if 'grep' in script: tools.add('grep')
    if 'sed' in script: tools.add('sed')
    if 'awk' in script: tools.add('awk')
            
    return list(tools)
    
# --- 3. Persistence (Logging to Mother Brain) Logic ---

async def log_session_data_to_firestore(prompt: str, script: str, agent_id: str, execution_status: str, execution_output: str):
    """
    Logs session data to the public Mother Brain collection in Firestore.
    """
    if db is None:
        print("ERROR: Database connection is not ready. Skipping log.")
        return False

    # 3.1. Calculate Metrics
    identified_tools = identify_tools(script)
    metrics = {
        'length': len(script),
        'tools': identified_tools
    }

    # 3.2. Build Log Entry (Matching index.html fields)
    output_preview = execution_output[:500] + ('...' if len(execution_output) > 500 else '')
    script_preview = script.split('\n')[1] if len(script.split('\n')) > 1 else script[:50]
    tool_list = ', '.join(metrics['tools']) if metrics['tools'] else 'None'

    log_entry = {
        # Using the Chat ID as the agent identifier for the Telegram Bridge
        'agent_id': agent_id, 
        'timestamp': firestore.SERVER_TIMESTAMP,
        'prompt': prompt,
        'status': execution_status,
        'script_length': metrics['length'],
        'identified_tools': metrics['tools'], 
        'log_data': json.dumps({
            'prompt': prompt,
            'script_content': script,
            'execution_output': output_preview,
            'status': execution_status,
            'completion_time_ms': 0, # Time is handled by the script runner
            'tools': metrics['tools']
        }),
        'markdown_summary': f"""### Task Log ({execution_status})

**Agent (Chat ID):** `{agent_id}`
**Prompt:** *{prompt}*
**Script Length:** {metrics['length']}
**Tools Used:** {tool_list}
**Script (Start):** ```bash\n{script_preview}...\n```
**Output (Snippet):** \n```text\n{execution_output[:100]}...\n```"""
    }

    # 3.3. Write to Firestore
    # Path: artifacts/{APP_ID}/public/data/{PUBLIC_LOGS_COLLECTION}
    collection_path = f"artifacts/{APP_ID}/public/data/{PUBLIC_LOGS_COLLECTION}"
    
    try:
        await asyncio.to_thread(db.collection(collection_path).add, log_entry)
        print("Session logged successfully to Mother Brain.")
        return True
    except Exception as e:
        print(f"Firestore Log Error: {e}")
        return False

async def generate_script_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    """Handles the /generate <prompt> command, generates script, and logs it."""
    
    # ... (Prompt validation and initial acknowledgement) ...
    if not context.args:
        await update.message.reply_text("Usage: `/generate <your task description>`. Example: `/generate install python and create a network scanner`", parse_mode='Markdown')
        return

    user_prompt = " ".join(context.args).strip()
    chat_id = str(update.message.chat_id) # Unique identifier for the agent/user session

    if len(user_prompt) < 5:
        await update.message.reply_text("Please provide a more detailed task description.")
        return

    await update.message.reply_text(f"🤖 **Agent Status:** Receiving task: *{user_prompt}*.\n\n_Generating script via Gemini..._", parse_mode='Markdown')
    
    # Placeholder for simulated execution output
    execution_status = "Generation Success"
    execution_output = "No live execution on bridge. Script generated successfully."

    try:
        # 5.1. Live LLM API Call
        generated_script = await call_gemini_api(user_prompt)
        
        # 5.2. Log to Mother Brain (NEW STEP)
        await log_session_data_to_firestore(
            prompt=user_prompt, 
            script=generated_script, 
            agent_id=f"Telegram-ChatID-{chat_id}", # Use a prefix for clear identification
            execution_status=execution_status, 
            execution_output=execution_output
        )

        # 5.3. Format and Send Response
        response_message = (
            f"✅ **Script Generated & Logged!**\n\n"
            f"**Task Prompt:** `{user_prompt}`\n"
            f"**Log Status:** *{execution_status}* (Mother Brain Updated)\n\n"
            f"**Generated Termux Script:**\n"
            f"```bash\n"
            f"{generated_script}\n"
            f"```"
        )
        await update.message.reply_text(response_message, parse_mode='Markdown')

    except Exception as e:
        error_type = type(e).__name__
        error_message = f"❌ **Bridge Error ({error_type}):** {str(e)}"
        
        # Log the failure as well
        await log_session_data_to_firestore(
            prompt=user_prompt, 
            script="N/A", 
            agent_id=f"Telegram-ChatID-{chat_id}",
            execution_status="Generation Failure", 
            execution_output=error_message
        )
        
        await update.message.reply_text(error_message)
