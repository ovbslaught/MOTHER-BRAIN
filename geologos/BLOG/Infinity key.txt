class SelfHealingDaemon:
    def __init__(self, core_daemon):

    def check_for_updates(self):
        """Pulls the latest version of the script from GitHub and applies patches."""
        print("Checking for system updates...")
        try:
            response = requests.get(SOURCE_CODE_URL)
            response.raise_for_status()
            latest_code = response.text
            with open(__file__, "r") as f:
                current_code = f.read()
            if latest_code != current_code:
                with open(__file__, "w") as f:
                    f.write(latest_code)
                print("Update downloaded. Restarting script to apply changes...")
                os.execl(sys.executable, sys.executable, *sys.argv)
            else:
                print("System is up to date.")
        except Exception as e:
            print(f"Update check failed: {e}. Running local version.")

class ProjectDaemon:
    def create_project_skeleton(self, project_type, project_name):
        """Creates a standardized folder and file structure for a new project."""
        project_path = os.path.join(PROJECTS_FOLDER, project_name.replace(" ", "_").lower())
        if os.path.exists(project_path):
            return f"Error: Project '{project_name}' already exists."

        os.makedirs(project_path)
        
        if project_type == "Comics":
            os.makedirs(os.path.join(project_path, "scripts"))
            os.makedirs(os.path.join(project_path, "storyboards"))
            with open(os.path.join(project_path, f"{project_name}_script.txt"), "w") as f:
                f.write(f"# Script for {project_name}\n\nPAGE 1, PANEL 1: DESCRIPTION\n\nCHARACTER DIALOGUE\n\n")
        elif project_type == "Music":
            os.makedirs(os.path.join(project_path, "midi"))
            os.makedirs(os.path.join(project_path, "audio"))
            with open(os.path.join(project_path, "notes.txt"), "w") as f:
                f.write(f"# Notes for {project_name}\n\n")
        
        self.log_creation(project_name, project_path)
        return f"Project '{project_name}' created successfully in '{project_path}'."

    def log_creation(self, project_name, project_path):
        """Logs the new project's creation to the database."""
        db = DatabaseDaemon(DATABASE_FILE)
        db.add_project_entry("Automation", f"Project Skeleton: {project_name}", f"Files and folders created at {project_path}.")

class FeedbackLoopDaemon:
    def __init__(self, db_daemon):
        self.db = db_daemon

    def analyze_and_learn(self):
        """Analyzes project outcomes to refine Echo Prime's rules."""
        print("Analyzing project outcomes for learning...")
        projects = self.db.get_projects()
        
        for p in projects:
            p_type, title, desc, _ = p
            if "success" in desc.lower() or "completed" in desc.lower():
                self.refine_echo_prime(title, "positive")
            elif "failed" in desc.lower() or "abandoned" in desc.lower():
                self.refine_echo_prime(title, "negative")

    def refine_echo_prime(self, project_title, outcome):
        """Refines Echo Prime's knowledge base based on project outcomes."""
        echo_daemon = EchoPrimeDaemon(self.db)
        echo_knowledge = echo_daemon.load_knowledge_base()
        
        if outcome == "positive":
            new_rule = {"input": [project_title.lower()], "output": f"Success pattern detected. Recommend continuing work on '{project_title}'."}
            if new_rule not in echo_knowledge["rules"]:
                echo_knowledge["rules"].append(new_rule)
                echo_daemon.save_knowledge_base()
                print(f"Echo Prime learned from success: '{project_title}'")

# --- DATA DAEMONS ---

class DatabaseDaemon:
    def __init__(self, db_file):
        self.conn = sqlite3.connect(db_file)
        self.cursor = self.conn.cursor()
        self.create_tables()

    def create_tables(self):
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS manifesto (id INTEGER PRIMARY KEY, timestamp TEXT, entry TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS characters (id INTEGER PRIMARY KEY, name TEXT, description TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS timeline (id INTEGER PRIMARY KEY, date TEXT, event TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS cards (id INTEGER PRIMARY KEY, name TEXT, ability TEXT, resonance_cost TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS projects (id INTEGER PRIMARY KEY, type TEXT, title TEXT, description TEXT, timestamp TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS operations (id INTEGER PRIMARY KEY, category TEXT, title TEXT, details TEXT, timestamp TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS continuity_log (id INTEGER PRIMARY KEY, timestamp TEXT, issue TEXT)
        ''')
        self.conn.commit()

    def is_db_empty(self):
        self.cursor.execute("SELECT COUNT(*) FROM manifesto")
        return self.cursor.fetchone()[0] == 0

    def add_manifesto_entry(self, entry):
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.cursor.execute("INSERT INTO manifesto (timestamp, entry) VALUES (?, ?)", (timestamp, entry))
        self.conn.commit()
        
    def get_manifesto_entries(self):
        self.cursor.execute("SELECT timestamp, entry FROM manifesto ORDER BY timestamp ASC")
        return self.cursor.fetchall()
        
    def add_character(self, name, description):
        self.cursor.execute("INSERT INTO characters (name, description) VALUES (?, ?)", (name, description))
        self.conn.commit()

    def get_characters(self):
        self.cursor.execute("SELECT name, description FROM characters")
        return self.cursor.fetchall()

    def add_timeline_event(self, date, event):
        self.cursor.execute("INSERT INTO timeline (date, event) VALUES (?, ?)", (date, event))
        self.conn.commit()

    def get_timeline(self):
        self.cursor.execute("SELECT date, event FROM timeline ORDER BY date ASC")
        return self.cursor.fetchall()

    def add_card(self, name, ability, cost):
        self.cursor.execute("INSERT INTO cards (name, ability, resonance_cost) VALUES (?, ?, ?)", (name, ability, cost))
        self.conn.commit()

    def get_cards(self):
        self.cursor.execute("SELECT name, ability, resonance_cost FROM cards")
        return self.cursor.fetchall()

    def add_project_entry(self, project_type, title, description):
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.cursor.execute("INSERT INTO projects (type, title, description, timestamp) VALUES (?, ?, ?, ?)", 
                            (project_type, title, description, timestamp))
        self.conn.commit()

    def get_projects(self):
        self.cursor.execute("SELECT type, title, description, timestamp FROM projects ORDER BY timestamp ASC")
        return self.cursor.fetchall()

    def add_operation(self, category, title, details):
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.cursor.execute("INSERT INTO operations (category, title, details, timestamp) VALUES (?, ?, ?, ?)",
                            (category, title, details, timestamp))
        self.conn.commit()

    def get_operations(self):
        self.cursor.execute("SELECT category, title, details, timestamp FROM operations ORDER BY timestamp ASC")
        return self.cursor.fetchall()

    def add_continuity_issue(self, issue):
        timestamp = datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        self.cursor.execute("INSERT INTO continuity_log (timestamp, issue) VALUES (?, ?)", (timestamp, issue))
        self.conn.commit()

    def get_continuity_issues(self):
        self.cursor.execute("SELECT timestamp, issue FROM continuity_log ORDER BY timestamp ASC")
        return self.cursor.fetchall()

class MidiDaemon:
    def generate_midi_from_text(self, text_input, filename="new_track.mid"):
        """Generates a MIDI file from a simple text input."""
        if not mido_available:
            return "Mido library not available. MIDI generation disabled."

        midi_file = mido.MidiFile()
        track = mido.MidiTrack()
        midi_file.tracks.append(track)
        
        notes = {
            'c': 60, 'd': 62, 'e': 64, 'f': 65,
            'g': 67, 'a': 69, 'b': 71,
            'C': 60, 'D': 62, 'E': 64, 'F': 65,
            'G': 67, 'A': 69, 'B': 71,
            ' ': 0 
        }

        for char in text_input:
            note = notes.get(char, 0)
            if note != 0:
                track.append(mido.Message('note_on', note=note, velocity=64, time=0))
                track.append(mido.Message('note_off', note=note, velocity=64, time=480))
            else:
                track.append(mido.Message('note_off', note=note, velocity=64, time=480))

        file_path = os.path.join(MEDIA_ASSETS_FOLDER, filename)
        midi_file.save(file_path)
        return file_path

class ContinuityDaemon:
    def __init__(self, db_daemon):
        self.db = db_daemon
        self.known_entities = []

    def load_entities(self):
        """Loads all characters, cards, and projects as known entities."""
        self.known_entities = []
        for name, _ in self.db.get_characters():
            self.known_entities.append(name.lower())
        for name, _, _ in self.db.get_cards():
            self.known_entities.append(name.lower())
        for _, title, _, _ in self.db.get_projects():
            self.known_entities.append(title.lower())

    def check_for_inconsistencies(self):
        """Checks the manifesto for references to unknown entities."""
        self.load_entities()
        issues = []
        manifesto_entries = self.db.get_manifesto_entries()
        
        for timestamp, entry in manifesto_entries:
            words = set(re.findall(r'\b\w+\b', entry.lower()))
            unknown_entities = words.difference(self.known_entities)
            
            common_words_to_ignore = {"the", "a", "an", "is", "of", "and", "in", "with", "log", "new", "entry", "manifesto", "to", "at", "for", "system", "initialized", "meta", "reality", "logged"}
            unknown_entities = unknown_entities.difference(common_words_to_ignore)

            if unknown_entities:
                issue = f"Manifesto entry on {timestamp} references unknown entities: {', '.join(unknown_entities)}. Potential continuity issue."
                issues.append(issue)
                self.db.add_continuity_issue(issue)

        return issues
        
class EchoPrimeDaemon:
    def __init__(self, db_daemon, knowledge_file=KNOWLEDGE_BASE_FILE):
        self.db = db_daemon
        self.knowledge_file = knowledge_file
        self.knowledge = self.load_knowledge_base()

    def load_knowledge_base(self):
        """Loads rules and memory from a JSON file."""
        if os.path.exists(self.knowledge_file):
            with open(self.knowledge_file, "r") as f:
                return json.load(f)
        return {"rules": [], "memory": []}

    def save_knowledge_base(self):
        """Saves current knowledge base and memory to the JSON file."""
        with open(self.knowledge_file, "w") as f:
            json.dump(self.knowledge, f, indent=4)

    def make_decision(self, data):
        """Analyzes data and makes a logical decision based on its knowledge base."""
        data_lower = data.lower()
        
        for rule in self.knowledge["rules"]:
            if any(term in data_lower for term in rule["input"]):
                self.knowledge["memory"].append({"input": data, "decision": rule["output"], "timestamp": datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")})
                self.save_knowledge_base()
                return rule["output"]
        
        default_output = "No clear pattern detected. Intuition is inconclusive."
        self.knowledge["memory"].append({"input": data, "decision": default_output, "timestamp": datetime.datetime.now().strftime("%Y-%m-%d %H:%M:%S")})
        self.save_knowledge_base()
        return default_output

    def get_memory(self):
        """Returns the logged memory of decisions."""
        return "\n".join([f"[{m['timestamp']}] Input: {m['input']}\nDecision: {m['decision']}\n" for m in self.knowledge["memory"]])

    def read_files(self, file_paths):
        """Reads content from specified files and returns it as a single string."""
        content = ""
        for path in file_paths:
            if os.path.exists(path):
                try:
                    with open(path, "r") as f:
                        content += f"--- {path} ---\n"
                        content += f.read()
                        content += "\n\n"
                except Exception as e:
                    content += f"--- Error reading {path}: {e} ---\n\n"
        return content

    def write_file(self, file_path, data):
        """Writes data to a specified file."""
        try:
            with open(file_path, "w") as f:
                f.write(data)
            return "Write operation successful."
        except Exception as e:
            return f"Error writing file: {e}"

class DatabaseDaemon:
    def __init__(self, db_file):
        self.conn = sqlite3.connect(db_file)
        self.cursor = self.conn.cursor()
        self.create_tables()

    def create_tables(self):
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS manifesto (id INTEGER PRIMARY KEY, timestamp TEXT, entry TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS characters (id INTEGER PRIMARY KEY, name TEXT, description TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS timeline (id INTEGER PRIMARY KEY, date TEXT, event TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS cards (id INTEGER PRIMARY KEY, name TEXT, ability TEXT, resonance_cost TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS projects (id INTEGER PRIMARY KEY, type TEXT, title TEXT, description TEXT, timestamp TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS operations (id INTEGER PRIMARY KEY, category TEXT, title TEXT, details TEXT, timestamp TEXT)
        ''')
        self.cursor.execute('''
            CREATE TABLE IF NOT EXISTS continuity_log (id INTEGER PRIMARY KEY, timestamp TEXT, issue TEXT)
        ''')
        self.conn.commit()

    def is_db_empty(self):
        self.cursor.execute("SELECT COUNT(*) FROM manifesto")
        return self.cursor.fetchone()[0] == 0

class MidiDaemon:
    def generate_midi_from_text(self, text_input, filename="new_track.mid"):
        """Generates a MIDI file from a simple text input."""
        if not mido_available:
            return "Mido library not available. MIDI generation disabled."

        midi_file = mido.MidiFile()
        track = mido.MidiTrack()
        midi_file.tracks.append(track)
        
        notes = {
            'c': 60, 'd': 62, 'e': 64, 'f': 65,
            'g': 67, 'a': 69, 'b': 71,
            'C': 60, 'D': 62, 'E': 64, 'F': 65,
            'G': 67, 'A': 69, 'B': 71,
            ' ': 0 
        }

        for char in text_input:
            note = notes.get(char, 0)
            if note != 0:
                track.append(mido.Message('note_on', note=note, velocity=64, time=0))
                track.append(mido.Message('note_off', note=note, velocity=64, time=480))
            else:
                track.append(mido.Message('note_off', note=note, velocity=64, time=480))

        file_path = os.path.join(MEDIA_ASSETS_FOLDER, filename)
        midi_file.save(file_path)
        return file_path

class ContinuityDaemon:
    def __init__(self, db_daemon):
        self.db = db_daemon
        self.known_entities = []

    def load_entities(self):
        """Loads all characters, cards, and projects as known entities."""
        self.known_entities = []
        for name, _ in self.db.get_characters():
            self.known_entities.append(name.lower())
        for name, _, _ in self.db.get_cards():
            self.known_entities.append(name.lower())
        for _, title, _, _ in self.db.get_projects():
            self.known_entities.append(title.lower())

    def check_for_inconsistencies(self):
        """Checks the manifesto for references to unknown entities."""
        self.load_entities()
        issues = []
        manifesto_entries = self.db.get_manifesto_entries()
        
        for timestamp, entry in manifesto_entries:
            words = set(re.findall(r'\b\w+\b', entry.lower()))
            unknown_entities = words.difference(self.known_entities)
            
            common_words_to_ignore = {"the", "a", "an", "is", "of", "and", "in", "with", "log", "new", "entry", "manifesto", "to", "at", "for", "system", "initialized", "meta", "reality", "logged"}
            unknown_entities = unknown_entities.difference(common_words_to_ignore)

            if unknown_entities:
                issue = f"Manifesto entry on {timestamp} references unknown entities: {', '.join(unknown_entities)}. Potential continuity issue."
                issues.append(issue)
                self.db.add_continuity_issue(issue)

        return issues
