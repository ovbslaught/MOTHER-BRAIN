from datetime import datetime

class MemoryGuard:
    """Rust-style ownership tracking in Python"""
    
    def __init__(self):
        self.allocations = {}
        self.total_alloc = 0
        self.total_freed = 0
        print("✓ MemoryGuard initialized (Rust-style ownership)")
    
    def alloc(self, name, size_bytes):
        mem_id = f"{name}_{datetime.utcnow().timestamp()}"
        self.allocations[mem_id] = {
            "name": name,
            "size_bytes": size_bytes,
            "alloc_time": datetime.utcnow().isoformat()
        }
        self.total_alloc += size_bytes
        return mem_id
    
    def dealloc(self, mem_id):
        if mem_id in self.allocations:
            size = self.allocations[mem_id]["size_bytes"]
            self.total_freed += size
            del self.allocations[mem_id]
    
    def get_stats(self):
        active_bytes = sum(a["size_bytes"] for a in self.allocations.values())
        return {
            "active_allocations": len(self.allocations),
            "active_mb": active_bytes / 1e6,
            "total_alloc_mb": self.total_alloc / 1e6,
            "total_freed_mb": self.total_freed / 1e6,
            "leak_check": len(self.allocations) == 0
        }
    
    def check_leaks(self):
        if self.allocations:
            print(f"⚠️  Memory leak: {len(self.allocations)} active allocations")
            for info in self.allocations.values():
                print(f"  - {info['name']}: {info['size_bytes']/1e6:.2f} MB")
            return True
        return False
EOPYTHON

import json
from datetime import datetime
from pathlib import Path

class MotherBrain:
    """Session snapshots + rehydration protocol"""
    
    def __init__(self):
        self.session_id = f"SESSION_{datetime.now().strftime('%Y%m%d_%H%M%S')}"
        self.session_log = {
            "session_id": self.session_id,
            "start_time": datetime.utcnow().isoformat(),
            "operations": []
        }
        print(f"✓ Mother Brain Protocol active: {self.session_id}")
    
    def log_operation(self, op_type, data):
        self.session_log["operations"].append({
            "timestamp": datetime.utcnow().isoformat(),
            "type": op_type,
            "data": data
        })
    
    def save_session(self):
        session_path = Path(f"../logs/motherbrain_{self.session_id}.json")
        with open(session_path, "w") as f:
            json.dump(self.session_log, f, indent=2)
        print(f"✓ Session saved: {session_path}")
        return session_path
    
    @staticmethod
    def rehydrate_session(session_path):
        """Restore from saved session"""
        with open(session_path) as f:
            session = json.load(f)
        print(f"✓ Rehydrated session: {session['session_id']}")
        return session
EOPYTHON
